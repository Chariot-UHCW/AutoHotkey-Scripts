Hotkey MessageCentreGUIKey, MessageCentreGUI

MessageCentreGUI(*) {
    Log("-- Enter Pre-OP Outcome GUI --", 1)
    MyGui := Gui(, "Enter Pre-Op Outcome") ; Using MyGui for clarity
    
    MyGui.AddText("", "Reply")
    
    ; 1. Add the DropDownList and link it to the 'UpdatePreview' function
    DDL := MyGui.AddDropDownList("w300 Choose1 vReply", [
        "APPROVED - Pre-Op Booked",
        "DENIED - Has Future Pre-Op",
        "DENIED - Had Past Pre-Op",
        "DENIED - Not a refresh"
    ])
    DDL.OnEvent("Change", UpdatePreview)

    ; 2. Assign the Edit box to a variable (PreviewEdit) so we can modify it
    PreviewEdit := MyGui.AddEdit("w300 r3 ReadOnly", "") 
    
    MyGui.AddButton("Default w300 xm y+24", "OK").OnEvent("Click", EnterOutcomeExe)
    
    ; Set the initial preview text based on the first item
    UpdatePreview() 
    
    MyGui.Show("AutoSize Center")

    ; --- Internal Functions ---

    UpdatePreview(*) {
        ; Logic to determine what the text should look like
        Selected := DDL.Text
        PreviewText := ""

        if (Selected = "APPROVED - Pre-Op Booked")
            PreviewText := "yes test"
        else if (InStr(Selected, "DENIED"))
            PreviewText := "denied test"
        
        ; Update the Edit box
        PreviewEdit.Value := PreviewText
    }

    EnterOutcomeExe(*) {
        Log("Running Pre-Op Outcome", 2)
        fields := MyGui.Submit()
        ; No need to Destroy here if you use Submit, but usually helpful
        
        Sleep(500)
        ; Send your logic here using PreviewEdit.Value or fields.Reply
        Send(PreviewEdit.Value)
    }
}
